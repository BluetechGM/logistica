CREATE TABLE compras_so.teste (
    id SERIAL PRIMARY KEY,
    placa VARCHAR(20),
    odometro_atual VARCHAR(50),
    odometro_prox_troca VARCHAR(50),
    data_prox_manutencao TIMESTAMP,
    cidade VARCHAR(100),
    criado_por VARCHAR(100),
    servico TEXT,
    status VARCHAR(50) DEFAULT 'Pendente',
    observacoes TEXT,
    ativo BOOLEAN DEFAULT true,
    filial_id VARCHAR(20),
    id_solicitacao VARCHAR(100),
    id_solicitante VARCHAR(50),
    nome_solicitante VARCHAR(100),
    link_orcamento_pdf TEXT,
    link_fotos_orcamento TEXT,
    link_video_orcamento TEXT,
    link_imagem_orcamento TEXT,
    tipo_incidente VARCHAR(100),
    nfe_servico_pdf TEXT,
    nfe_produtos_pdf TEXT,
    foto_comprovante TEXT,
    recnum INTEGER,
    lancado_winthor BOOLEAN DEFAULT false,
    data_abertura TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    data_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
